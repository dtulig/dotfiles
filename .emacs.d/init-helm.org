* Helm Loading and Configuration

** Implementation.

#+BEGIN_SRC emacs-lisp
  (require 'helm)
  (require 'helm-config)

  (global-set-key (kbd "C-c h") 'helm-command-prefix)
  (global-unset-key (kbd "C-x c"))

  (when (executable-find "curl")
    (setq helm-google-suggest-use-curl-p t))

  (helm-mode 1)
#+END_SRC
